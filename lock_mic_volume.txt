# Require Administrator privileges
$isAdmin = ([Security.Principal.WindowsPrincipal][Security.Principal.WindowsIdentity]::GetCurrent()).IsInRole([Security.Principal.WindowsBuiltInRole]::Administrator)
if (-not $isAdmin) {
    if ($PSCommandPath) {
        Write-Host "Requesting Administrator privileges..."
        Start-Process powershell.exe "-NoProfile -ExecutionPolicy Bypass -File `"$PSCommandPath`"" -Verb RunAs
    } else {
        Write-Warning "This script requires Administrator privileges!"
        Write-Host "Because you are running via 'irm | iex', it cannot auto-elevate."
        Write-Host "Please open 'PowerShell as Administrator' and run the command again."
    }
    exit
}

$workDir = "C:\phwyverysad"

# Main Menu
Clear-Host
Write-Host "============================="
Write-Host " Microphone Volume Manager "
Write-Host "============================="
Write-Host "1. Install"
Write-Host "2. Uninstall"
Write-Host "============================="
$mainChoice = Read-Host "Please enter your choice (1 or 2)"

if ($mainChoice -eq '1') {
    # Step 1: Create directory
    if (!(Test-Path $workDir)) {
        New-Item -ItemType Directory -Force -Path $workDir | Out-Null
        Write-Host "Created folder: $workDir"
    }

    # Step 2: Download file
    $zipUrl = "https://is.gd/NZJRo6"
    $zipPath = "$workDir\lock_mic_volume.zip"
    Write-Host "Downloading files from $zipUrl ..."
    Invoke-WebRequest -Uri $zipUrl -OutFile $zipPath

    # Step 3: Extract file
    Write-Host "Extracting files..."
    Expand-Archive -Path $zipPath -DestinationPath $workDir -Force

    # Step 4: Move contents from inner folder to root workDir and delete inner folder
    $innerFolder = "$workDir\lock_mic_volume"
    if (Test-Path $innerFolder) {
        Move-Item -Path "$innerFolder\*" -Destination $workDir -Force
        Remove-Item -Path $innerFolder -Recurse -Force
    }

    # Step 5: Delete the zip file
    Remove-Item -Path $zipPath -Force
    Write-Host "Cleaned up zip file."

    # Step 6: Percentage Menu
    Write-Host "`n============================="
    Write-Host " Select Volume Percentage "
    Write-Host "============================="
    Write-Host "1. 100%"
    Write-Host "2. 75%"
    Write-Host "3. 50%"
    Write-Host "4. 25%"
    Write-Host "============================="
    $volChoice = Read-Host "Please enter your choice (1-4)"

    $targetFolder = ""
    switch ($volChoice) {
        '1' { $targetFolder = "100%" }
        '2' { $targetFolder = "75%" }
        '3' { $targetFolder = "50%" }
        '4' { $targetFolder = "25%" }
        default { 
            Write-Host "Invalid choice. Exiting..."
            Start-Sleep -Seconds 3
            exit 
        }
    }

    # Step 7: Run the batch file, wait 3 seconds, then force close
    $batFolder = "$workDir\$targetFolder"
    $batFile = "$batFolder\Run_atomatically.bat"

    if (Test-Path $batFile) {
        Write-Host "Executing script: $batFile ..."
        Set-Location -Path $batFolder
        
        # Run process and capture its info
        $batProcess = Start-Process -FilePath "cmd.exe" -ArgumentList "/c `"$batFile`"" -PassThru
        
        Write-Host "Waiting 3 seconds..."
        Start-Sleep -Seconds 3
        
        # Force close if it's still running
        if ($batProcess -and -not $batProcess.HasExited) {
            Write-Host "Force closing the script..."
            Stop-Process -Id $batProcess.Id -Force -ErrorAction SilentlyContinue
        }
    } else {
        Write-Host "Error: Cannot find $batFile"
    }

    # Step 8: Clean up the main working directory
    Set-Location -Path "C:\"
    Write-Host "Cleaning up working directory ($workDir)..."
    Remove-Item -Path $workDir -Recurse -Force -ErrorAction SilentlyContinue
    
    Write-Host "Installation completed successfully."
    Start-Sleep -Seconds 3
}
elseif ($mainChoice -eq '2') {
    Write-Host "Starting uninstallation process..."
    
    # Step 1: Kill nircmdc.exe process
    Write-Host "Terminating nircmdc.exe process..."
    taskkill /IM nircmdc.exe /F 2>$null

    # Step 2: Force delete system and startup files
    $startupFolder = [Environment]::GetFolderPath('Startup')
    $filesToDelete = @(
        "C:\Windows\lock_mic_vol.bat",
        "C:\Windows\hide_cmd_window2.vbs",
        "C:\Windows\nircmdc.exe",
        "$startupFolder\start_lock_mic_vol.bat"
    )

    foreach ($file in $filesToDelete) {
        if (Test-Path $file) {
            Remove-Item -Path $file -Force -ErrorAction SilentlyContinue
            Write-Host "Deleted: $file"
        } else {
            Write-Host "File not found (already deleted): $file"
        }
    }
    
    Write-Host "Uninstallation completed successfully."
    Start-Sleep -Seconds 3
}
else {
    Write-Host "Invalid choice. Exiting..."
    Start-Sleep -Seconds 3
}
